\section{Requirement Analysis}

\subsection{Basic Features}
\begin{frame}{Basic Features}

\insertlg{thinking cat.jpg}

\begin{itemize}
    \item \textbf{Effortless Photo Management}
    
    Organize your photos seamlessly by time, location, and custom tags. Easily browse and retrieve memories with structured categorization.

    \item \textbf{Advanced Search Capabilities}

    Instantly find your photos based on content, date, location, or tags. Use AI-powered search to recognize faces, objects, and scenes within your photos.

    \item \textbf{Built-in Photo Editing Tools}

    Enhance your photos with powerful editing tools, including cropping, filters, adjustments, and retouching optionsâ€”all within the app.  

    
    
\end{itemize}

\end{frame}

\begin{frame}{Basic Features (Cont.)}

\insertlg{thinking cat.jpg}

\begin{itemize}
    \item \textbf{Memory Timeline \& Video Generation}
        
    Relive special moments with automatically generated memory timelines. The system can create video highlights using your best photos and clips.
        
    \item \textbf{Seamless Social Sharing \& Collaboration}

    Share your favorite moments effortlessly with family and friends. Collaborate on shared albums, comment on photos, and react with emojis.
\end{itemize}

\end{frame}

\subsection{Non-Functional Features}

\begin{frame}{Non-Functional Features}

\insertlg{thinking cat.jpg}

\begin{itemize}
    
    \item \textbf{Usability} 
    
    The application offers an intuitive, multilingual, and accessible interface with responsive design for seamless use across devices.

    \item \textbf{Performance}
    
    Ensures fast response times, scalability, and efficient caching for handling large albums and high user concurrency.

    \item \textbf{Security}
    
    Implements robust authentication, role-based access control, and encryption to protect user data and ensure secure access.

\end{itemize}
    
\end{frame}

\begin{frame}{Non-Functional Features (Cont.)}

\insertlg{thinking cat.jpg}

\begin{itemize}
    \item \textbf{Safety \& Privacy}
    
    Complies with data protection regulations, provides data deletion options, and ensures secure handling of sensitive information.

    \item \textbf{Maintainability}
    
    Features modular code, comprehensive documentation, automated testing, and CI/CD pipelines for easy maintenance and updates.
\end{itemize}
\end{frame}

            }
        \end{column}
    \end{columns}

\end{frame}

\begin{frame}{How to Collect and Process Data}

\insertlg{thinking cat.jpg}

\begin{itemize}
    \item \textbf{Image Metadata Extraction}
    
    Automatically extract EXIF data using libraries (e.g., ExifTool, Python Pillow).

    \item \textbf{AI Tagging \& Classification }
    
    Use OpenCV and GPT (or other AI/ML services) for face recognition, object detection, and tagging.

    \item \textbf{Photo Editing Metadata}
    
    Implement edit history and version control to enable rollbacks or undo changes.
    \item \textbf{Search Indexes}
    
    Build search indexes in databases (e.g., ElasticSearch) to enable fast retrieval based on time, location, people, objects, and tags.
\end{itemize}

\end{frame}

\begin{frame}{Data Storage}

\insertlg{thinking cat.jpg}

\begin{itemize}
    \item Use \textbf{MINIO} for object storage (photos, videos).

    \item Use \textbf{MySQL/PostgreSQL} for relational data (user accounts, tags, album relationships).
    
    \item \textbf{Backup and disaster recovery plans} to protect user data.
\end{itemize}

\end{frame}
